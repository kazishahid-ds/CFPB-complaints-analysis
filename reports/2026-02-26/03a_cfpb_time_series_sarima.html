3.1.2 Differencing Fix

In[5]:
# Stationarity test
adf_result = adfuller(ts)
print(f"ADF p-value: {adf_result[1]:.3f} {'Stationary' if adf_result[1]<0.05 else 'Non-stationary'}")

ADF p-value: 0.999 Non-stationary
In[22]:
# First difference to fix non-stationarity
ts_diff = ts.diff().dropna()
adf = adfuller(ts_diff)
print(f"Differenced ADF p-value: {adf[1]:.4f}")
ts_diff.plot(title='Differenced Series')
plt.show()

3.1.3 Simpler SARIMA

In[]:
# Simpler model
model = SARIMAX(
    ts,
    order=(1, 1, 0), # AR only, 1 difference
    seasonal_order=(1, 1, 0, 4), # Seasonal AR only
    trend='c' # Add constant for trend
)
fitted = model.fit(disp=False)
print(fitted.summary())

3.1.4 Forecast & Plot

In[26]:
# Forecast next 4 quarters
forecast_obj = fitted.get_forecast(steps=4)
forecast_mean = forecast_obj.predicted_mean
forecast_ci = forecast_obj.conf_int()

# Plot
fig, ax = plt.subplots(figsize=(12, 5))
ts.plot(ax=ax, label='Observed', color='steelblue')
forecast_mean.plot(ax=ax, label='Forecast', color='tomato')
ax.fill_between(
    forecast_ci.index,
    forecast_ci.iloc[:, 0],
    forecast_ci.iloc[:, 1],
    color='tomato', alpha=0.2, label='95% CI'
)
ax.set_title('SARIMA(1,1,0)(1,1,0,4) Baseline Forecast')
ax.legend()
plt.tight_layout()
plt.savefig('../reports/figures/sarima_forecast.png', dpi=150)
